CREATE TABLE IF NOT EXISTS item_details(
id int AUTO_INCREMENT,
item_1 varchar(20) NOT NULL,
price_1 int,
item_2 varchar(20) NOT NULL,
price_2 int,
item_3 varchar(20) NOT NULL,
price_3 int,
item_4 varchar(20) NOT NULL,
price_4 int,
item_5 varchar(20) NOT NULL,
price_5 int,
item_6 varchar(20) NOT NULL,
price_6 int,
item_7 varchar(20) NOT NULL,
price_7 int,
item_8 varchar(20) NOT NULL,
price_8 int,
item_9 varchar(20) NOT NULL,
price_9 int,
item_10 varchar(20) NOT NULL,
price_10 int,
restaurant_id int NOT NULL,
PRIMARY KEY(id),
FOREIGN KEY(restaurant_id) REFERENCES restaurant(restaurant_id))
ENGINE=INNODB;
